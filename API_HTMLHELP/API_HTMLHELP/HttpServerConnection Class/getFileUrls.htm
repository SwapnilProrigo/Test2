HttpServerConnection Class

getFileUrls(ArrayList, UrlType)

Supply an ArrayList of file ids which returns each file url vault location in an ArrayList.

C#:

HttpServerConnection cnx = IomFactory.CreateHttpServerConnection(url, db, user, pw);
Item login_result = cnx.Login();
ArrayList list = new ArrayList();
Item file = innovator.newItem("File", "get");
file.setProperty("filename", "customers.txt");
file = file.apply();
string fileid = file.getID();
list.Add(fileid);
ArrayList urls =cnx.getFileUrls(list, UrlType.None);
return inn.newResult(urls[0].ToString());

VB:

Dim cnx As HttpServerConnection = IomFactory.CreateHttpServerConnection(url, db, user, pw)
Dim login_result As Item = cnx.Login()
Dim inn As Innovator = Me.getInnovator()
Dim list As New ArrayList()
Dim file As Item = innovator.newItem("File", "get")
file.setProperty("filename", "customers.txt")
file = file.apply()
Dim fileid As String = file.getID()
list.Add(fileid)
Dim urls As ArrayList = cnx.getFileUrls(list, UrlType.None)
Return inn.newResult(urls(0).ToString())

Notes:

Possible URLType for enumeration - None or SecurityToken.
