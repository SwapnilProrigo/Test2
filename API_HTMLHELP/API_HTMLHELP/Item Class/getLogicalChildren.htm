Item Class

getLogicalChildren()

Returns Item object with the nodeList containing logical nodes that are children of the this.node.

In this example, a query is built to get all Parts that have a cost of “100” AND make_buy=Buy – OR – have a cost of "1000" and make_buy=Make.

C#:

Item qry = this.newItem("Part", "get");
Item logicItm = qry.newOR();
Item logicAND1 = logicItm.newAND();
logicAND1.setProperty("cost", "100");
logicAND1.setProperty("make_buy", "Buy");
Item logicAND2 = logicItm.newAND();
logicAND2.setProperty("cost", "1000");
logicAND2.setProperty("make_buy", "Make");
Item retItm = qry.apply();

Next - get logical "children" of the query node (which is the AND conditions and remove them)

Item lchildren = qry.getLogicalChildren();

for( int i = 0; i < lchildren.getItemCount(); i++ )
{
  Item lchild = lchildren.getItemByIndex(i);
  qry.removeLogical(lchild);
}

VB:

Dim qry As Item = Me.newItem("Part, "get")
Dim logicItm As Item = qry.newOR()
Dim logicAND1 As Item = logicItm.newAND()
logicAND1.setProperty("cost", "100")
logicAND1.setProperty("make_buy", "Buy")
Dim logicAND2 As Item = logicItm.newAND()
logicAND2.setProperty("cost", "1000")
logicAND2.setProperty("make_buy", "Make")
Dim retItm As Item = qry.apply()

Next - get logical "children" of the query node (which is the AND conditions and remove them)

Dim lchildren As Item = qry.getLogicalChildren()
For i As Integer = 0 To lchildren.getItemCount() - 1
 Dim lchild As Item = lchildren.getItemByIndex(i)
 qry.removeLogical(lchild)
Next

Notes:

Returns item object with the nodeList containing logical nodes that are direct children of the this.node. The returned item shares its dom with this item.
